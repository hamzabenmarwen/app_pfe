\let\Originalincludegraphics\includegraphics
\newcommand{\missinggraphic}[1]{%
  \fbox{%
    \parbox[c][0.22\textheight][c]{0.86\linewidth}{%
      \centering
      \textbf{Illustration à insérer}\\[2mm]
      \small Fichier attendu : \texttt{\detokenize{#1}}%
    }%
  }%
}
\newcommand{\smartincludegraphics}[2][]{%
  % Keep placeholders only for truly missing files while still checking ./diagrams.
  \IfFileExists{#2}{%
    \Originalincludegraphics[#1]{#2}%
  }{%
    \IfFileExists{diagrams/#2}{%
      \Originalincludegraphics[#1]{diagrams/#2}%
    }{%
      \missinggraphic{#2}%
    }%
  }%
}
\renewcommand{\includegraphics}[2][]{%
  \smartincludegraphics[#1]{#2}%
}
